הגדרות — מה הסקריפט צריך לדעת מראש

// SECTION 1: DEFINITIONS
javascript — section 1
// SECTION 1: DEFINITIONS const SPECIAL_CHARS = [ { char: '!', name: 'סימן קריאה !' }, { char: '@', name: 'כרוכית @' }, { char: '#', name: 'סולמית #' }, { char: '$', name: 'דולר $' }, { char: '%', name: 'אחוז %' }, { char: '*', name: 'כוכבית *' }, ]; const PLACEHOLDER = '§'; const BASE_PASSWORD = `Abc1${PLACEHOLDER}defgh`; const makePassword = (char) => BASE_PASSWORD.replace(PLACEHOLDER, char); let _captured = null;

מה הסקשן הזה עושה?

זהו ה"הכנה" של הסקריפט — כאן מוגדרות כל הנקודות הקבועות שהסקריפט ישתמש בהן לאורך כל הריצה. ממש כמו שלפני ניסוי במעבדה מכינים את כל הכלים, הכאן מגדירים את רשימת התווים לבדיקה, את מבנה הסיסמה, ואת "הכיס" שישמור את הבקשה שנלכדה.

פירוט כל קטע

  • SPECIAL_CHARS — מערך (רשימה) של אובייקטים. כל אובייקט מייצג תו מיוחד אחד ומכיל שני שדות: char (התו עצמו) ו-name (שמו בעברית לצורך הדוח). הסקריפט יעבור על כל האיברים ברשימה הזו אחד אחד.
  • PLACEHOLDER — תו ייחודי (§) שמשמש כסמן מיקום בתבנית הסיסמה. נבחר תו נדיר כדי שלא ייתקל בטעות בשם שדה אחר.
  • BASE_PASSWORD — תבנית הסיסמה: Abc1§defgh. הסיסמה עומדת בדרישות נפוצות (אות גדולה, אות קטנה, ספרה) ומשאירה "חור" — ה-§ — שיוחלף בכל פעם בתו מיוחד אחר.
  • makePassword(char) — פונקציה קטנה שמקבלת תו ומחזירה סיסמה מוכנה. למשל: makePassword('!') יחזיר Abc1!defgh.
  • _captured — משתנה שמתחיל כ-null (ריק). הוא ימולא אוטומטית ברגע שהסקריפט "יתפוס" בקשת רשת שהאפליקציה שולחת. כל שאר הסקריפט מסתמך על המשתנה הזה.
אנלוגיה

דמיינו שאתם מכינים 6 מעטפות לשליחה, ועל כל מעטפה תכתבו שם שונה. לפני שמתחילים, מכינים: רשימת שמות (SPECIAL_CHARS), תבנית המעטפה (BASE_PASSWORD), ותיבת דואר ריקה שתחכה לקלט (_captured).

למה זה נדרש?

הגדרת הכל מראש בצורה מסודרת מאפשרת לשנות בקלות את רשימת התווים הנבדקים או את תבנית הסיסמה מבלי לגעת בשאר הלוגיקה. הפרדת "המה" (הנתונים) מה"איך" (הלוגיקה) היא עיקרון מרכזי בכתיבת קוד תחזוקתי.

מה הסקשן הזה עושה?

הסקשן הזה "מחדיר" קוד מרגל לתוך פונקציית הרשת המובנית של הדפדפן. כשהאפליקציה שולחת בקשת fetch (דרך הרשת לשרת), הסקריפט שלנו נכנס לפניה, מצלם את כל הפרטים, ואז מעביר את הבקשה המקורית הלאה — כך שהאפליקציה לא יודעת שמישהו הציץ.

שלבי הפעולה

  • שמירת המקור: לפני הכל, שומרים את window.fetch המקורית תחת השם _originalFetch. זה קריטי — בלעדיו לא נוכל לשלוח בקשות בהמשך.
  • החלפת הפונקציה: window.fetch מוחלפת בגרסה שלנו. מעכשיו כל קריאה ל-fetch תעבור דרכנו.
  • חילוץ מידע: הקוד מוציא URL, גוף הבקשה (body) ושיטת הבקשה (method).
  • סינון: לכידה מתבצעת רק עבור POST, PUT או PATCH (שיטות שמכילות נתונים כמו סיסמה). בקשות GET מתעלמות.
  • שמירה ב-_captured: הפרטים נשמרים כדי שניתן יהיה לשחזר את הבקשה בהמשך.
  • הודעה בקונסול: הדפסה כחולה מציינת שנלכדה בקשה.
  • העברה למקורי: בסוף, הבקשה האמיתית ממשיכה לשרת ללא שינוי.
אנלוגיה — דוור מצלם מכתבים

דמיינו דוור שעובד בבניין משרדים. רגיל, הוא מעביר מכתבים מהשולח לנמען. הסקריפט שלנו הוא כמו דוור חדש שנכנס לעמדה: הוא מצלם כל מכתב שעובר בידיו, ממשיך להעביר אותו לנמען — אבל עכשיו יש לו עותק. מאוחר יותר הוא יוכל לשלוח גרסאות שונות של אותו מכתב.

מושג טכני — Monkey Patching

Monkey Patching הוא טכניקה שבה מחליפים בזמן ריצה פונקציה מובנית בגרסה מותאמת אישית. השם "קוף" מרמז על שינוי "צעקני" ולא מסודר — אבל בהקשר של DevTools לבדיקות זו שיטה נפוצה ולגיטימית. חשוב: ה-Monkey Patching פועל רק בדף הנוכחי ונמחק ברגע שמרעננים את הדפדפן.

למה זה נדרש?

לא ניתן "לאסור" על אפליקציה לשלוח בקשות בצורה ישירה מבחוץ — אבל אפשר לשים את עצמנו בתווך. כך אנחנו מקבלים עותק מדויק של הבקשה האמיתית, כולל headers ו-body, בלי להתקין כלים חיצוניים.

יירוט XHR — כיסוי הטכנולוגיה הישנה

// SECTION 3: XHR INTERCEPTOR

// SECTION 3: XHR INTERCEPTOR const _xhrOpen = XMLHttpRequest.prototype.open; const _xhrSetHeader = XMLHttpRequest.prototype.setRequestHeader; const _xhrSend = XMLHttpRequest.prototype.send; XMLHttpRequest.prototype.open = function (method, url) { this._m = method.toUpperCase(); this._u = url; this._h = {}; return _xhrOpen.apply(this, arguments); }; XMLHttpRequest.prototype.setRequestHeader = function (name, value) { this._h[name] = value; return _xhrSetHeader.apply(this, arguments); }; XMLHttpRequest.prototype.send = function (body) { if (body && ['POST', 'PUT', 'PATCH'].includes(this._m)) { _captured = { type : 'xhr', url : this._u, method : this._m, headers : { ...this._h }, bodyRaw : typeof body === 'string' ? body : null, }; } return _xhrSend.apply(this, arguments); };

מה הסקשן הזה עושה?

בעולם הדפדפנים יש שתי שיטות עיקריות לשליחת בקשות רשת: fetch (מודרנית) ו-XMLHttpRequest (ישנה יותר, מכונה XHR). הסקשן הקודם כיסה את fetch. הסקשן הזה עושה בדיוק אותו דבר — אבל עבור XHR. כך הסקריפט מוכן לתפוס בקשות מכל אפליקציה, ישנה כחדשה.

מדוע XHR מורכב יותר לפצל?

ב-fetch פונקציה אחת מטפלת בהכל. ב-XHR קיימות שלוש פונקציות נפרדות שעובדות ברצף:

  • open(method, url) — מגדירה את היעד ושיטת הבקשה. הסקריפט שומר את ה-method ב-this._m וה-URL ב-this._u ומאתחל מילון ריק לכותרות (this._h).
  • setRequestHeader(name, value) — מוסיפה כותרת אחת. הסקריפט מצלם כל כותרת למילון this._h כדי שיוכל לשחזר אותן בהמשך.
  • send(body) — שולחת את הבקשה עם הגוף. כאן מתבצעת הלכידה: אם השיטה היא POST/PUT/PATCH, כל המידע שנצבר (URL, method, headers, body) נשמר ב-_captured.
אנלוגיה — ריגול בשלושה שלבים

ב-fetch, הדוור מקבל מכתב ממולא לגמרי. ב-XHR, השולח בונה את המכתב בהדרגה: קודם כותב כתובת יעד (open), אחר כך מדביק בולים (setRequestHeader), ולבסוף שם את התוכן בתוך המעטפה (send). הריגול שלנו צריך "להציץ" בכל שלב ולאסוף את החלקים.

למה שני יירוטים?

לא ניתן לדעת מראש באיזו שיטה האפליקציה משתמשת. אפליקציות רבות (במיוחד ישנות יותר, או כאלו שמשתמשות בספריות כמו jQuery) עדיין שולחות דרך XHR. אם לא נכסה את שניהם, הסקריפט לא יעבוד חצי הפעמים.

מושג טכני — prototype

ב-JavaScript, prototype הוא "תבנית" שממנה יורשות כל המופעים של אובייקט. שינוי XMLHttpRequest.prototype.send אומר: "כל XHR שייווצר בעתיד יפעיל את הגרסה שלי". זה בדיוק כמו לשנות את ה-DNA של כל החרקים מסוג מסוים — כולם שתשקף את השינוי.

_cloneInit — העתקה בטוחה של הגדרות הבקשה

// SECTION 4: _cloneInit
javascript — section 4
// SECTION 4: _cloneInit function _cloneInit(init) { try { return JSON.parse(JSON.stringify(init)); } catch { return { ...init }; } }

מה הסקשן הזה עושה?

הפונקציה _cloneInit יוצרת העתק מלא ועצמאי של אובייקט הגדרות הבקשה (init). זה אולי נראה טריוויאלי, אבל זוהי אחת הנקודות הכי עדינות ומסוכנות בתכנות JavaScript.

הבעיה — העתקה שטחית

ב-JavaScript, כשכותבים const copy = obj, לא נוצר עותק — שניהם מצביעים לאותו אובייקט בזיכרון. אם לאחר מכן מבצעים copy.body = "new password", גם obj.body ישתנה! זה אומר שהבקשה המקורית תישלח עם הסיסמה ששונתה — דבר שעלול לשבור את הבדיקה.

הפתרון — העתקה עמוקה בשני שלבים

  • JSON.stringify(init) — הופך את האובייקט למחרוזת טקסט. בשלב זה, הקשר לאובייקט המקורי נקטע.
  • JSON.parse(...) — בונה אובייקט חדש לגמרי מהמחרוזת. הוא עצמאי לחלוטין מהמקור.
  • catch fallback: אם הסריאליזציה נכשלת (למשל, אם יש בתוך init פונקציות שלא ניתן להמיר ל-JSON), מבוצעת העתקה שטחית ({ ...init }) — לא מושלמת, אבל טובה יותר מכלום.
אנלוגיה — מסמך נייר לעומת PDF

"העתקה שטחית" היא כמו להצביע על נייר ולהגיד "זה העותק שלי". אם מישהו ישנה את הנייר המקורי — העותק שלכם ישתנה גם. JSON.stringify + JSON.parse הוא כמו לצלם את הנייר לPDF ולהדפיס אותו מחדש — כעת יש לכם עמוד עצמאי לחלוטין.

למה זה נדרש?

הסקריפט שולח 6 בקשות, כל אחת עם סיסמה שונה. אם לא נשכפל את init כראוי, שינוי הסיסמה בבקשה השנייה יגרום לבאג בבקשה הראשונה — ותוצאות הבדיקה יהיו שגויות.

מה הסקשן הזה עושה?

זהו "לב" הסקריפט. לאחר שתפסנו את הבקשה המקורית, אנחנו צריכים לשנות רק את שדה הסיסמה — ולהשאיר הכל אחר ללא שינוי. הפונקציות הללו מבצעות חיפוש חכם וגמיש של שדה הסיסמה, ללא תלות בשם המדויק של השדה.

_replacePassword — הלוגיקה הראשית

הפונקציה עובדת בסדר עדיפויות:

  • מקרה 1 — גוף ריק: אם אין body, מחזירים אותו כמות שהוא.
  • מקרה 2 — JSON: מנסים לפרסר את ה-body כ-JSON. אם הצלחנו, מפעילים _deepReplace לחיפוש ושינוי רקורסיבי, ואז מחזירים את התוצאה כמחרוזת JSON.
  • מקרה 3 — Form Encoded: אם ה-JSON נכשל וה-body מכיל =, כנראה שמדובר בפורמט password=abc&username=user. ה-RegEx מחפש מפתחות בשם password/pass/pwd/newPassword/new_password ומחליף את הערך.
  • מקרה 4 — פורמט לא מזוהה: מחזירים כמות שהוא.

_deepReplace — חיפוש רקורסיבי

הפונקציה "צוללת" לתוך אובייקטים מקוננים (nested) ומחפשת שדות שמכילים את המילים password, pass, pwd, newPass, או secret בשמם. כשנמצא שדה כזה — ערכו מוחלף. לאחר מכן ממשיכים רקורסיבית (כלומר, הפונקציה קוראת לעצמה) עבור כל ערך מקונן.

אנלוגיה — בדיקת רכוש לפני מעבר

דמיינו שאתם מועברים לבית חדש ורוצים לוודא שאין בו שום סכין. אתם בודקים כל מגירה, כל ארגז, כל כיס של כל בגד. _deepReplace עושה בדיוק את זה — בודק כל שדה, כל אובייקט מקונן, כדי לוודא שאין סיסמה שנשמרה עמוק בתוך המבנה.

מושג טכני — JSON vs Form Encoded

JSON: {"username":"user","password":"abc123"} — מבנה נתונים עשיר, תומך בהיררכיה.
Form Encoded: username=user&password=abc123 — פורמט פשוט, נפוץ בטפסים ישנים.
הסקריפט תומך בשניהם כי כל שרת יכול לדרוש פורמט שונה.

מושג טכני — Recursion (רקורסיה)

רקורסיה היא כשפונקציה קוראת לעצמה. ב-_deepReplace, כל ערך שהוא אובייקט מועבר שוב לאותה פונקציה — כך ניתן לטפל במבנים עמוקים כמה רמות שיהיו, ללא צורך לדעת מראש את עומק המבנה.

_sendFetch — שחזור הבקשה דרך fetch

// SECTION 6: _sendFetch
javascript — section 6
// SECTION 6: _sendFetch async function _sendFetch(password) { const newInit = _cloneInit(_captured.init); newInit.body = _replacePassword(_captured.bodyRaw, password); const res = await _originalFetch(_captured.url, newInit); const text = await res.text().catch(() => ''); return { ok: res.ok, status: res.status, body: text }; }

מה הסקשן הזה עושה?

_sendFetch לוקחת את הבקשה שנלכדה, מחליפה בה את הסיסמה בסיסמה הנבדקת, ושולחת אותה לשרת — ואז מחכה לתשובה ומחזירה אותה.

שלבי הפעולה

  • שכפול ה-init: _cloneInit(_captured.init) — יוצר העתק בטוח שלא ישפיע על הבקשות הבאות.
  • החלפת הסיסמה: _replacePassword מוכנסת לגוף ה-init המשוכפל.
  • שליחה דרך המקורי: חשוב מאוד — משתמשים ב-_originalFetch ולא ב-window.fetch. אחרת הסקריפט יתפוס את עצמו בלולאה אינסופית!
  • קריאת תשובה: res.text() קורא את גוף התשובה כטקסט. ה-.catch(() => '') מוודא שגם אם הקריאה נכשלת, לא קורסת הפונקציה.
  • החזרת תוצאה: מחזיר אובייקט עם שלושה שדות: ok (האם הצלחה), status (קוד HTTP), ו-body (תוכן התשובה).
אנלוגיה — שליחת מכתב עם כתובת שונה

הדוור הצלם כעת שולח גרסאות שונות של אותו מכתב — הכל זהה חוץ מ"שם הנמען" (הסיסמה). הוא שומר עותק של כל תשובה שמגיעה חזרה ומדווח לנו.

מושג טכני — async/await

פונקציות רשת לוקחות זמן. async מסמנת פונקציה שיכולה "לחכות" לתוצאה. await אומר "עצור כאן וחכה שהבקשה תחזור — אבל אל תחסום את שאר הדפדפן". בלי async/await, הסקריפט יריץ את כל הלולאה מיד בלי לחכות לתשובות.

_sendXHR — שחזור הבקשה דרך XHR עם Promise

// SECTION 7: _sendXHR
javascript — section 7
// SECTION 7: _sendXHR function _sendXHR(password) { return new Promise((resolve, reject) => { const xhr = new XMLHttpRequest(); xhr.open(_captured.method, _captured.url); for (const [name, value] of Object.entries(_captured.headers)) { try { xhr.setRequestHeader(name, value); } catch {} } xhr.onload = () => resolve({ ok: xhr.status < 400, status: xhr.status, body: xhr.responseText }); xhr.onerror = () => reject(new Error('XHR network error')); xhr.ontimeout = () => reject(new Error('XHR timeout')); xhr.timeout = 8000; const newBody = _replacePassword(_captured.bodyRaw, password); xhr.send(newBody); }); }

מה הסקשן הזה עושה?

_sendXHR מבצעת בדיוק את מה שעושה _sendFetch — אך עבור בקשות שנלכדו כ-XHR. ההבדל הטכני המשמעותי הוא שפונקציה זו עוטפת את ה-XHR (שהוא מבוסס-callback) בתוך Promise, כך שניתן להשתמש בה עם await בדיוק כמו _sendFetch.

שלבי הפעולה

  • new Promise(resolve, reject): יוצר "הבטחה" — אובייקט שמייצג ערך עתידי. resolve מופעל כשהבקשה מסתיימת בהצלחה, reject — כשיש שגיאה.
  • פתיחת חיבור: xhr.open עם ה-method וה-URL שנלכדו.
  • שחזור headers: לולאה על כל הכותרות שנשמרו ב-_captured.headers. ה-try/catch מסביב מגן מפני כותרות שהדפדפן לא מרשה לשלוח ידנית.
  • הגדרת handlers:
    • onload — מופעל כשהתשובה הגיעה (גם אם קוד שגיאה). ok נקבע לפי סטטוס מתחת ל-400.
    • onerror — בעיית רשת (אין קישוריות).
    • ontimeout — הבקשה לקחה יותר מ-8 שניות.
  • שליחה: _replacePassword מחליפה את הסיסמה ואז שולחת את הבקשה.
מושג טכני — Promise ו-callback

XHR עובד ב-callback style: "כשתסיים, תקרא לפונקציה הזו". זה לא תואם עם async/await ישירות. ה-Promise הוא מעטפת שמתרגמת את ה-callback לסגנון מודרני. פשוט: Promise אומר "אחכה לתשובה ואחזיר אותה כמו ערך רגיל — ולא דרך callback".

למה timeout של 8 שניות?

אם השרת לא עונה (הפסקת רשת, שרת תקוע), הלולאה הראשית תיתקע לנצח. 8 שניות הוא "timeout סביר" — מספיק לשרתים איטיים, אך לא מבזבז זמן על בקשות מתות.

testChars — הפונקציה הראשית: הרצת כל הבדיקות

// SECTION 8: testChars — MAIN FUNCTION
javascript — section 8
// SECTION 8: testChars — MAIN FUNCTION async function testChars() { if (!_captured) { console.error('❌ לא נלכדה בקשה עדיין!'); return; } const results = []; for (const { char, name } of SPECIAL_CHARS) { const password = makePassword(char); let response; try { if (_captured.type === 'fetch') { response = await _sendFetch(password); } else { response = await _sendXHR(password); } let result, note; if (response.ok) { result = '✅ עובד'; note = 'הסיסמא התקבלה'; } else if (response.status === 400) { result = '❌ נחסם — validation'; note = 'השרת דחה (400)'; } else if (response.status === 500) { result = '❌ נחסם — שגיאת שרת'; note = 'שגיאת שרת (500)'; } else if ([401,403].includes(response.status)) { result = '⚠️ token פג'; note = `${response.status}`; } else { result = `⚠️ ${response.status}`; note = ''; } results.push({ תו: char, שם: name, סיסמא: password, status: response.status, תוצאה: result, הערה: note }); } catch (err) { results.push({ תו: char, שם: name, סיסמא: password, status: 'ERR', תוצאה: '❌ שגיאת רשת', הערה: err.message }); } await new Promise(r => setTimeout(r, 600)); } console.table(results); }

מה הסקשן הזה עושה?

זוהי הפונקציה שמפעילים ידנית בקונסול (testChars()). היא מרכזת את כל שאר הסקשנים לכדי תהליך בדיקה אחד: עוברת על כל תו מיוחד, שולחת בקשה, מפרשת את התשובה ומדפיסה טבלת תוצאות.

שלבי הפעולה

  • בדיקת Guard: אם _captured עדיין null — כלומר עוד לא נלכדה בקשה — הפונקציה עוצרת ומציגה שגיאה. חשוב לא לדלג על שלב הלוגין לפני הרצת הפונקציה.
  • לולאת for...of: עוברת על כל אחד מ-6 האיברים ב-SPECIAL_CHARS. בכל איטרציה, יוצרת סיסמה, שולחת בקשה ומקבלת תשובה.
  • בחירת שיטת שליחה: לפי _captured.type — אם "fetch" מפעיל _sendFetch, אחרת _sendXHR.
  • פרשנות התשובה: על בסיס קוד ה-HTTP, נקבעת תוצאה מילולית לדוח.
  • try/catch: אם הבקשה קורסת (timeout, רשת נפלה), שגיאת הרשת נשמרת כתוצאה ולא עוצרת את הלולאה.
  • השהייה של 600ms: מחכה 0.6 שניות בין כל בקשה. מונע Rate Limiting (חסימה של השרת עקב בקשות רבות מהירות) ונותן לשרת זמן נשימה.
  • console.table: מדפיס את כל התוצאות בפורמט טבלה קריאה בקונסול.
אנלוגיה — מנהל QA ששולח בודקים

testChars הוא כמו מנהל QA שיש לו רשימת מקרים לבדיקה. הוא שולח בודק אחד בכל פעם, מחכה שיחזור עם דוח, רושם את הממצא, ורק אז שולח את הבא. בסוף — מרכז הכל בטבלה אחת.

נקודת קשב — 401/403 הם לא כשל של התו

אם מתקבל 401 או 403, כנראה שה-authentication token פג תוקף בזמן הריצה. יש להתחבר מחדש, להדביק את הסקריפט שוב, ולהריץ testChars() מחדש. זו לא בעיה של התו המיוחד עצמו.

מדוע await new Promise(r => setTimeout(r, 600))?

זוהי דרך אלגנטית ל"שינה אסינכרונית". setTimeout הוא callback-based, אז עוטפים אותו ב-Promise. await גורם ללולאה לחכות 600ms לפני המשך — בלי לחסום את הדפדפן.

הצהרת משתנה — Scope, Hoisting ו-TDZ

שלוש מילות המפתח יוצרות "קופסאות זיכרון", אך עם הבדלים קריטיים במי רואה אותן, ומה קורה לפני שהן מוגדרות.

// ─── SCOPE ─── תחום הגישה function example() { if (true) { var x = 10; // var "בורח" מה-if — function scope let y = 20; // let נכלא ב-{ } — block scope const z = 30; // const נכלא ב-{ } — block scope } console.log(x); // ✅ 10 (var "דלף" החוצה) console.log(y); // ❌ ReferenceError (let כלוא) } // ─── HOISTING ─── "הרמה" לראש הפונקציה console.log(a); // → undefined (var "הורם" אבל לא הערך!) var a = 5; console.log(b); // ❌ ReferenceError — TDZ (ראו הסבר) let b = 5; // מה JavaScript עשה מתחת למכסה עם var: // var a; ← הוקם לראש הסקופ (ריק) // console.log(a); ← undefined // a = 5; ← השמה קורית בשורה המקורית // ─── TDZ — Temporal Dead Zone ─── // let ו-const "קיימים" מתחילת ה-block, אבל אסור לגשת אליהם לפני ה-let/const { // ← כאן מתחיל ה-TDZ של z console.log(z); // ❌ ReferenceError: Cannot access 'z' before initialization const z = 99; // ← כאן ה-TDZ מסתיים console.log(z); // ✅ 99 } // ─── CONST — לא "קפואה" לגמרי ─── const arr = [1, 2, 3]; arr.push(4); // ✅ מותר — שינוי תוכן המערך arr = [5, 6]; // ❌ TypeError — לא ניתן להחליף ה-reference
אנלוגיה

var = מפתח שנמצא בכיס המעיל — גישה מכל מקום בחדר (פונקציה).
let = מפתח שנמצא בתיבה ספציפית — רק מי שבחדר הספציפי ({ }) יכול להשתמש בו.
const = כתובת חקוקה בשיש — לא ניתן לשנות לאן מצביעים, אך ניתן לשנות את הדבר שבכתובת.
TDZ = הספר בספריה כבר "רשום" בשם שלכם, אבל טרם הגיע — לא ניתן לקחת אותו עד שיגיע.

מלכודות נפוצות

1. var בתוך לולאה — כל הסיבובים חולקים אותה משתנה אחת. לעולם השתמשו ב-let בלולאות.
2. const obj = {} — ניתן לשנות תכונות, לא ניתן להחליף את האובייקט כולו.
3. TDZ נוגע גם לפרמטרים ברירת מחדל: function f(a = b, b = 2) — TDZ error כי b לא הוגדר עדיין!

Closure Scope & Functions
Closure — פונקציה שזוכרת את הסביבה שנולדה בה

Closure נוצר כשפונקציה "רואה" ומשתמשת במשתנים מה-scope שבו הוגדרה — גם אחרי שה-scope ההוא "נסגר". זהו אחד המושגים החשובים ביותר ב-JavaScript.

// ─── דוגמה בסיסית ─── function createCounter() { let count = 0; // משתנה פרטי ב-scope של createCounter return function() { // הפונקציה הפנימית היא ה-closure count++; // "רואה" ו"זוכרת" את count — גם אחרי שcreateCounter סיים return count; }; } const counter = createCounter(); // createCounter רץ ו"נסגר" counter(); // → 1 (count עדיין חי בזיכרון!) counter(); // → 2 counter(); // → 3 // ─── Factory Function — שימוש מעשי ─── function makeMultiplier(factor) { return n => n * factor; // "זוכרת" את factor } const double = makeMultiplier(2); const triple = makeMultiplier(3); double(5); // → 10 (factor = 2) triple(5); // → 15 (factor = 3) // ─── Closure שומר על פרטיות ─── function createBankAccount(initialBalance) { let balance = initialBalance; // ← פרטי! לא ניתן לגישה ישירה מבחוץ return { deposit: amount => { balance += amount; }, withdraw: amount => { balance -= amount; }, getBalance: () => balance }; }
אנלוגיה

הכנתם קפה בבית (= הסביבה), ויצאתם לטיול עם הכוס. הכוס "זוכרת" שיצאה מהבית ויש בה קפה — גם כשהבית נעול מאחוריכם. ה-closure היא הכוס: פונקציה שלוקחת איתה את מה שהיה ב-scope שנולדה בו.

בסקריפט שלנו

כל הפונקציות (סקשנים 4–8) הן closure מעל _captured ו-_originalFetch. הן הוגדרו ב-scope הגלובלי ו"זוכרות" את שני המשתנים — גם כשנקראות מתוך לולאה בזמן הריצה. makePassword היא closure מעל BASE_PASSWORD ו-PLACEHOLDER.

Event Loop + async/await Asynchronous JavaScript
איך JavaScript "מחכה" בלי לקפוא — Event Loop

JavaScript הוא Single Threaded — שרשור יחיד בלבד. רק פעולה אחת מתבצעת בכל רגע. אז איך הוא שולח בקשת רשת וממשיך לרוץ בלי לחכות?

אנלוגיה

מסעדה עם מלצר אחד בלבד. הוא מקבל הזמנה ושולח למטבח — לא עומד ומחכה, ממשיך לקבל הזמנות אחרות. כשמנה מוכנה, הוא חוזר ומגיש אותה. זהו Event Loop. await אומר: "אחכה לפריט הזה לפני שאמשיך" — אבל המלצר עדיין לא חסום לשאר הלקוחות.

מלכודות נפוצות

1. await עובד רק בתוך async function — בלי async תקבלו SyntaxError.
2. await בתוך forEach לא ממתין כצפוי! השתמשו ב-for...of במקום.
3. async function תמיד מחזירה Promise — לא ניתן "לשלוף" ממנה ערך סינכרוני.

בסקריפט שלנו

async function testChars() — ללא async, ה-await _sendFetch() היה גורם ל-SyntaxError. await new Promise(r => setTimeout(r, 600)) = שינה אסינכרונית שלא חוסמת את הדפדפן בין בקשה לבקשה.

מלכודת — UnhandledPromiseRejection

Promise שנדחה (REJECTED) ואין לו .catch() או try/catch — גורם לאזהרה קריטית (ובNode.js מציאות קורס התהליך). תמיד טפלו בשגיאות!
