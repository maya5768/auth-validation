1. מה הבעיה?

הUI (הצד הלקוחי) מאשר סיסמא עם כל אחד מ-! @ # $ % * — כל הV מירוקים. אבל כשלוחצים שמירה, השרת מחזיר "משהו השתבש" במקום הצלחה. זה אומר שהשרת חוסם תווים מסוימים שה-UI לא יודע עליהם.

למה % בפרט חשוד?

% הוא תו ה-URL encoding. בכתובות URL ובנתוני טופס, %XX פירושו "תו עם קוד הקסדצימלי XX".
אם השרת מנסה לפענח את הסיסמא כ-URL ו-%ha אינו קוד hex תקין — השרת קורס בשגיאת parsing ומחזיר 500 במקום הודעת שגיאה ברורה.
תושםמדוע עלול להיחסם בשרתסיכון
!סימן קריאהבטוח ברוב הפורמטיםנמוך
@כרוכיתבטוח ברוב הפורמטיםנמוך
#סולמיתב-URL — סמן עוגן (fragment). עלול לקטוע את הנתיב.בינוני
$דולרבטוח ברוב הפורמטיםנמוך
%אחוזURL encoding prefix — %xx לא תקין שובר parsingגבוה ⚠️
*כוכביתwildcard בחלק מהשרתים, עלול לגרום ל-regex injectionבינוני

2. איך הסקריפט עובד?

  • יירוט (Intercept)
    הסקריפט "עוטף" את פונקציות fetch ו-XMLHttpRequest של הדפדפן. כשאת לוחצת שמירה, הבקשה עוברת דרכו — והוא שומר את כל הפרטים: כתובת ה-API, headers (כולל authentication token), וגוף הבקשה עם הסיסמא.
  • החלפת הסיסמא
    הסקריפט לוקח את הבקשה שנלכדה ומחליף רק את הסיסמא בסיסמא חדשה לכל בדיקה. כל שאר הנתונים (token, user ID וכו') נשארים זהים — כך הבקשה תקינה לחלוטין.
  • שליחה ישירה ל-API
    הסקריפט שולח כל בקשה ישירות לשרת (bypass מה-UI לגמרי) ומקבל את תגובת ה-HTTP: 200 = עבר, 400 = נדחה ב-validation, 500 = שגיאת שרת (תו שבור).
  • דו"ח תוצאות
    בסיום — טבלת תוצאות עם כל תו, ה-status code שקיבלנו, ואם עבר או נחסם.
⚠️ שים לב: הסקריפט שולח בקשות אמיתיות לשרת. אם תו מסוים עובר — הסיסמא שלך תשתנה לסיסמא של הבדיקה האחרונה שעברה.
שמרי בצד את הסיסמאות האפשריות לפני הריצה, או החליפי ידנית לאחר סיום הבדיקות.

3. הוראות שימוש שלב אחר שלב

  • הגיעי לדף "הגדרת הסיסמה שלך"
    כנסי לזרימה של שכחתי סיסמא עד שמגיעות לשדות "סיסמה חדשה" ו"אימות סיסמה".
  • פתחי DevTools — F12 → Console
    הקפידי שהלשונית Console פעילה.
  • הדביקי את הסקריפט כולו ← Enter
    אם Chrome חוסם הדבקה: הקלידי allow pasting ← Enter, אחר כך הדביקי. תקבלי הודעה ירוקה: "סקריפט בדיקת תווים מיוחדים — מוכן!"
  • מלאי ידנית סיסמא תקינה ולחצי שמירה
    לדוגמה: Abc1!defgh
    הסקריפט ילכוד את הבקשה ויציג: "📡 Fetch נלכד!"
    לא חשוב אם הסיסמא עוברת או לא — העיקר שהבקשה נשלחת.
  • הריצי: testChars()
    הקלידי בconsole: testChars() ← Enter
    הסקריפט ישלח 6 בקשות (כ-4 שניות) ויציג טבלת תוצאות.

4. הסקריפט — להעתקה

הדביקי את הכל בבת אחת — הסקריפט מפעיל את היירוט אוטומטית מיד עם ההדבקה. לאחר מכן רצי testChars() בלבד.
// ============================================================
// Special Characters Server Tester — ! @ # $ % *
// שלב א: הדבקי → שלב ב: שמרי פעם אחת → שלב ג: הריצי testChars()
// ============================================================

const SPECIAL_CHARS = [
  { char: '!', name: 'סימן קריאה !' },
  { char: '@', name: 'כרוכית @' },
  { char: '#', name: 'סולמית #' },
  { char: '$', name: 'דולר $' },
  { char: '%', name: 'אחוז %' },
  { char: '*', name: 'כוכבית *' },
];
const makePassword = (c) => `Abc1${c}defgh`;
let _captured = null;

// — יירוט fetch —
const _origFetch = window.fetch;
window.fetch = async function(input, init = {}) {
  const url = typeof input === 'string' ? input : input?.url;
  const body = init?.body ?? input?.body;
  const m = (init?.method ?? 'GET').toUpperCase();
  if (body && ['POST','PUT','PATCH'].includes(m)) {
    try { _captured = { type:'fetch', url, init: JSON.parse(JSON.stringify(init)), bodyRaw: typeof body==='string'?body:null }; } catch{};
    console.log('%c📡 נלכד! ', 'color:#0066cc;font-weight:bold', url);
  }
  return _origFetch.apply(this, arguments);
};

// — יירוט XHR —
const [_xO,_xH,_xS] = [XMLHttpRequest.prototype.open, XMLHttpRequest.prototype.setRequestHeader, XMLHttpRequest.prototype.send];
XMLHttpRequest.prototype.open = function(m,u){this._m=m.toUpperCase(); this._u=u; this._h={}; return _xO.apply(this,arguments);};
XMLHttpRequest.prototype.setRequestHeader = function(n,v){this._h[n]=v; return _xH.apply(this,arguments);};
XMLHttpRequest.prototype.send = function(body){
  if(body && ['POST','PUT','PATCH'].includes(this._m)){
    _captured={type:'xhr',url:this._u,method:this._m,headers:{...this._h},bodyRaw:typeof body==='string'?body:null};
    console.log('%c📡 XHR נלכד!','color:#0066cc;font-weight:bold',this._u);
  }
  return _xS.apply(this,arguments);
};

// — החלפת סיסמא בגוף הבקשה —
function _replacePw(body, pw) {
  if(!body) return body;
  try {
    const fix = (o) => {
      if(typeof o !== 'object' || !o) return o;
      const r = Array.isArray(o)?[]:{};
      for(const k of Object.keys(o)) r[k] = /password|pass|pwd|newPass/i.test(k) && typeof o[k]==='string' ? pw : fix(o[k]);
      return r;
    };
    return JSON.stringify(fix(JSON.parse(body)));
  } catch {
    return body.replace(/([&?]?(?:password|pass|pwd)[^&=]*)=([^&]*)/gi, (_,k) => `${k}=${encodeURIComponent(pw)}`);
  }
}

// — הפונקציה הראשית —
async function testChars() {
  if(!_captured){console.error('❌ לא נלכדה בקשה. לחצי שמירה קודם!'); return;}
  console.log('%c🚀 בודק 6 תווים...','color:#0066cc;font-weight:bold');
  const res = [];
  for (const {char, name} of SPECIAL_CHARS) {
    const pw = makePassword(char);
    console.log(`⏳ ${char} → ${pw}`);
    try {
      let r;
      if(_captured.type==='fetch') {
        const i = JSON.parse(JSON.stringify(_captured.init));
        i.body = _replacePw(_captured.bodyRaw, pw);
        const f = await _origFetch(_captured.url, i);
        r = {ok:f.ok, status:f.status, body: await f.text().catch(()=>'')};
      } else {
        r = await new Promise((resolve,reject)=>{
          const x=new XMLHttpRequest(); x.open(_captured.method,_captured.url);
          Object.entries(_captured.headers).forEach(([n,v])=>{try{x.setRequestHeader(n,v)}catch{}});
          x.onload=()=>resolve({ok:x.status<400, status:x.status, body:x.responseText});
          x.onerror=()=>reject(new Error('XHR error')); x.timeout=8000;
          x.send(_replacePw(_captured.bodyRaw,pw));
        });
      }
      const result = r.ok?'✅ עובד':r.status===500?'❌ שגיאת שרת 500':r.status===400?'❌ נדחה (400)':`⚠️ ${r.status}`;
      res.push({תו:char, שם:name, סיסמא:pw, status:r.status, תוצאה:result});
      console.log(` → ${result} (${r.status})`);
    } catch(e) { res.push({תו:char,שם:name,סיסמא:pw,status:'ERR',תוצאה:'❌ '+e.message}); }
    await new Promise(r=>setTimeout(r,600));
  }
  console.table(res);
  const ok=res.filter(r=>r.תוצאה.includes('✅')), bad=res.filter(r=>!r.תוצאה.includes('✅'));
  console.log(`%c✅ עובדים (${ok.length}): ${ok.map(r=>r.תו).join(' ')}`,'color:green;font-weight:bold');
  console.log(`%c❌ נחסמים (${bad.length}): ${bad.map(r=>r.תו).join(' ')}`,'color:red;font-weight:bold');
}

console.log('%c✅ מוכן! מלאי סיסמא → לחצי שמירה → הריצי: testChars()','color:green;font-weight:bold;font-size:13px');
